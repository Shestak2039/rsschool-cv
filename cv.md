function validParentheses(parens) {
  var counter = 0;
  for (var i = 0; i < parens.length; i++) {
    parens[i] == "(" ? counter++ : counter--;
    if (counter < 0) return false;
  }
  return counter == 0;
}

var uniqueInOrder = function(iterable) {
  var arr = [];
  if (iterable.length == 0) return [];
  arr.push(iterable[0]);
